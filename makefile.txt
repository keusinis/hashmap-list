CXX      = g++
CXXFLAGS = -std=c++17 -Wall -Wextra -pedantic

# (a) Build the module object file
hashmap.o: hashmap.h hashmap.cpp
	$(CXX) $(CXXFLAGS) -c hashmap.cpp -o hashmap.o

# (b) Build the demo executable
demo.o: hashmap.o demo.cpp
	$(CXX) $(CXXFLAGS) demo.cpp hashmap.o -o demo.o

# (c) Run the demo
demo: demo.o
	./demo.o

# (d) Build the test executable
test.o: hashmap.o test.cpp
	$(CXX) $(CXXFLAGS) test.cpp hashmap.o -o test.o

# (e) Run the tests (writes to log.txt)
test: test.o
	./test.o

# (f) Clean all generated artefacts
clean:
	rm -f *.o log.txt

# (g) Rebuild everything from scratch
rebuild: clean demo.o test.o
